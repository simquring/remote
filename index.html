export default { async fetch(request) { const cors = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "GET, POST, OPTIONS", "Access-Control-Allow-Headers": "Content-Type" }; if (request.method === "OPTIONS") { return new Response(null, { headers: cors }); } if (request.method !== "POST") { return new Response(JSON.stringify({ success: false, error: "Use POST" }), { headers: cors }); } try { const { url } = await request.json(); if (!url) { return new Response(JSON.stringify({ success: false, error: "URL kosong" }), { headers: cors }); } // 🔽 fetch video let videoRes; try { videoRes = await fetch(url); } catch { return new Response(JSON.stringify({ success: false, error: "Gagal fetch video" }), { headers: cors }); } if (!videoRes.ok) { return new Response(JSON.stringify({ success: false, error: "Video tidak bisa diakses" }), { headers: cors }); } const blob = await videoRes.blob(); // 🔽 upload simulasi ke videy let uploadRes; try { const formData = new FormData(); formData.append("file", blob, "video.mp4"); uploadRes = await fetch("https://videy.co/upload", { method: "POST", body: formData, headers: { "User-Agent": "Mozilla/5.0", "Origin": "https://videy.co", "Referer": "https://videy.co/" } }); } catch { return new Response(JSON.stringify({ success: false, error: "Upload gagal (kemungkinan diblok)" }), { headers: cors }); } const text = await uploadRes.text(); return new Response(JSON.stringify({ success: true, status: uploadRes.status, result: text.substring(0, 300) }), { headers: { ...cors, "Content-Type": "application/json" } }); } catch (err) { return new Response(JSON.stringify({ success: false, error: err.message }), { headers: cors }); } } };
